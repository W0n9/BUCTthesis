\buctsetup{

	% 论文的中文标题
	ctitle		= {BUCTthesis写作示例文档},

	ckeywords   = {论文,\LaTeX{},模板},

\begin{document}
	% 生成伪封面。实际的论文封面由学校统一印制，正式写作可删去。
